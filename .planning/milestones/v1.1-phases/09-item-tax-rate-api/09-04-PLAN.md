private async validateTaxRateExists(taxRateId: string): Promise<void> {
  try {
    await this.taxRateRepository.findByIdOrFail(taxRateId);
  } catch (error) {
    if (error instanceof ResourceNotFoundError) {
      throw new InvalidRequestError(ErrorCodes.TAX_RATE_NOT_FOUND, `The specified tax rate does not exist: ${taxRateId}`);
    }
    throw error;
  }
}
```

export class ItemUpdaterService implements IUpdaterService {
  constructor(
    private readonly itemRepository: ItemRepository,
    private readonly itemPolicy: ItemPolicy,
    private readonly accessControllerFactory: AccessControllerFactory,
  ) {}

  public async update(authUser: IUserDto, existing: IItemDto, updates: IItemDto): Promise<IItemDto> {
    const accessController = this.accessControllerFactory.create(this.itemPolicy);
    accessController.canUpdate(authUser, existing);
    const updatedItem = await this.itemRepository.update(updates);
    return updatedItem;
  }
}
```
