   ```typescript
   function calculateDaysRemaining(trialEnd: string): number {
     const end = new Date(trialEnd);
     const now = new Date();
     return Math.max(0, Math.ceil((end.getTime() - now.getTime()) / (1000 * 60 * 60 * 24)));
   }
   ```
