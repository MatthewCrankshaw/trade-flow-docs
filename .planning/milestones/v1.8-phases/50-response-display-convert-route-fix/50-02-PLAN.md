    Follow the `EstimateLostReasonCard` analog pattern. Named export function component with:

    ```typescript
    import { ThumbsUp, MessageSquare, ThumbsDown } from "lucide-react";
    import { formatDateTime } from "@/lib/date-helpers";
    import type { EstimateResponseSummary } from "@/types/estimate";

    const RESPONSE_TYPE_LABELS: Record<string, string> = {
      proceed: "Happy to Proceed",
      message: "Customer Message",
      decline: "Declined",
    };

    const RESPONSE_TYPE_ICONS: Record<string, React.ElementType> = {
      proceed: ThumbsUp,
      message: MessageSquare,
      decline: ThumbsDown,
    };

    const RESPONSE_TYPE_COLORS: Record<string, string> = {
      proceed: "text-green-600 dark:text-green-400",
      message: "text-amber-600 dark:text-amber-400",
      decline: "text-destructive",
    };

    const DECLINE_REASON_LABELS: Record<string, string> = {
      too_expensive: "Too expensive",
      going_with_someone_else: "Going with someone else",
      decided_not_to_do_work: "Decided not to do the work",
      just_getting_idea: "Just getting an idea of costs",
      timing_not_right: "Timing isn't right",
    };

    interface EstimateResponseCardProps {
      responseSummary: EstimateResponseSummary;
    }

    export function EstimateResponseCard({ responseSummary }: EstimateResponseCardProps) {
      const { lastResponseType, lastResponseAt, lastResponseMessage, declineReason } = responseSummary;

      if (!lastResponseType) return null;

      const Icon = RESPONSE_TYPE_ICONS[lastResponseType];
      const iconColor = RESPONSE_TYPE_COLORS[lastResponseType] ?? "";
      const label = RESPONSE_TYPE_LABELS[lastResponseType] ?? lastResponseType;

      return (
        <div className="flex flex-col gap-3">
          <div>
            <div className="flex items-center gap-2">
              {Icon && <Icon className={`h-4 w-4 ${iconColor}`} />}
              <span className="text-sm font-semibold">{label}</span>
            </div>
            {lastResponseAt && (
              <p className="text-xs text-muted-foreground mt-1">{formatDateTime(lastResponseAt)}</p>
            )}
          </div>
          {declineReason && (
            <p className="text-sm text-muted-foreground">
              Reason: {DECLINE_REASON_LABELS[declineReason] ?? declineReason}
            </p>
          )}
          {lastResponseMessage && (
            <p className="text-sm">&ldquo;{lastResponseMessage}&rdquo;</p>
          )}
        </div>
      );
    }
    ```
