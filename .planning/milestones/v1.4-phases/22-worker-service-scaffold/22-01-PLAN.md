    ```typescript
    import { NestFactory } from "@nestjs/core";
    import { Logger } from "@nestjs/common";
    import { Logger as PinoLogger } from "nestjs-pino";
    import { WorkerModule } from "@worker/worker.module";

    async function bootstrap() {
      const app = await NestFactory.createApplicationContext(WorkerModule, {
        bufferLogs: true,
      });

      app.useLogger(app.get(PinoLogger));

      const logger = new Logger("Worker");
      logger.log("Worker service started");

      process.on("SIGTERM", async () => {
        logger.log("SIGTERM received, shutting down gracefully");
        await app.close();
        process.exit(0);
      });

      process.on("SIGINT", async () => {
        logger.log("SIGINT received, shutting down gracefully");
        await app.close();
        process.exit(0);
      });
    }
    bootstrap();
    ```

    Key differences from main.ts:
    - Uses `createApplicationContext` NOT `NestFactory.create` (no HTTP server)
    - No `app.listen()`, no CORS, no ValidationPipe, no global pipes
    - Imports WorkerModule, not AppModule
    - Logger context is "Worker" for bootstrap messages
