```tsx
export function RoleActions({ targetUser, viewer }: RoleActionsProps) {
  const [showGrantDialog, setShowGrantDialog] = useState(false);
  const [showRevokeDialog, setShowRevokeDialog] = useState(false);

  // Only super users see role action buttons (per D-08)
  const viewerIsSuperUser = viewer.supportRoles?.some(
    (role) => role.roleName === "super_user"
  );
  if (!viewerIsSuperUser) return null;

  const targetHasNoSupportRole = !targetUser.supportRoles?.length;
  const targetIsAdmin = targetUser.supportRoles?.some(
    (role) => role.roleName === "admin" // or "support_administrator" -- check SupportRoleName enum values
  );
  const targetIsSuperUser = targetUser.supportRoles?.some(
    (role) => role.roleName === "super_user"
  );
  const isViewingSelf = viewer.id === targetUser.id;

  // Grant: shown only for customer users with no support role
  if (targetHasNoSupportRole) {
    return (
      <div className="mt-4">
        <Button variant="default" onClick={() => setShowGrantDialog(true)}
          aria-label={`Grant admin role to ${targetUser.name}`}>
          <ShieldPlus className="mr-2 h-4 w-4" />
          Grant Admin Role
        </Button>
        <GrantRoleDialog
          targetUserId={targetUser.id}
          targetUserName={targetUser.name}
          open={showGrantDialog}
          onOpenChange={setShowGrantDialog}
        />
      </div>
    );
  }

  // Revoke: shown for admin users, not self, not super users (per D-10, D-12)
  if (targetIsAdmin && !isViewingSelf && !targetIsSuperUser) {
    return (
      <div className="mt-4">
        <Button variant="destructive" onClick={() => setShowRevokeDialog(true)}
          aria-label={`Revoke admin role from ${targetUser.name}`}>
          <ShieldMinus className="mr-2 h-4 w-4" />
          Revoke Admin Role
        </Button>
        <RevokeRoleDialog
          targetUserId={targetUser.id}
          targetUserName={targetUser.name}
          open={showRevokeDialog}
          onOpenChange={setShowRevokeDialog}
        />
      </div>
    );
  }

  // Super user profile or own profile: no action buttons
  return null;
}
```

Note on role name values: Check the actual `SupportRoleName` enum values used in the API response. The backend may return `"super_user"` and `"support_administrator"` (or `"admin"`). Match the exact strings returned by the API's `IUserResponse.supportRoles[].roleName` field.
