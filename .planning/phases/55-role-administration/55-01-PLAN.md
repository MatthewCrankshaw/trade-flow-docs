```typescript
import { Injectable } from "@nestjs/common";
import { SupportRoleRepository } from "@user/repositories/support-role.repository";
import { SupportRoleName } from "@user/enums/support-role-name.enum";
import { UserRepository } from "@user/repositories/user.repository";
import { UserRetriever } from "@user/services/user-retriever.service";
import { IUserDto } from "@user/data-transfer-objects/user.dto";
import { AppLogger } from "@core/services/app-logger.service";
import { ForbiddenError } from "@core/errors/forbidden-error.error";
import { ErrorCodes } from "@core/errors/error-codes.enum";
import { InvalidRequestError } from "@core/errors/invalid-request-error.error";

@Injectable()
export class SupportRoleAssigner {
  private readonly logger = new AppLogger(SupportRoleAssigner.name);

  constructor(
    private readonly supportRoleRepository: SupportRoleRepository,
    private readonly userRepository: UserRepository,
    private readonly userRetriever: UserRetriever,
  ) {}

  public async grant(authUser: IUserDto, targetUserId: string): Promise<IUserDto> {
    const targetUser = await this.userRetriever.findByIdOrFail(targetUserId);

    // Reject if target already has any support role
    if (targetUser.supportRoleIds && targetUser.supportRoleIds.length > 0) {
      throw new InvalidRequestError(ErrorCodes.INVALID_REQUEST, "User already has a support role");
    }

    // Find the seeded Admin role (per D-02)
    const adminRole = await this.supportRoleRepository.findByRoleName(SupportRoleName.SUPPORT_ADMINISTRATOR);
    if (!adminRole) {
      throw new Error("Admin support role not found in database -- ensure seeder has run");
    }

    await this.userRepository.addSupportRoleId(targetUserId, adminRole.id);
    this.logger.log("Granted admin support role", { grantedBy: authUser.id, targetUserId, roleId: adminRole.id });

    return this.userRetriever.findByIdOrFail(targetUserId);
  }

  public async revoke(authUser: IUserDto, targetUserId: string): Promise<IUserDto> {
    // Self-revocation guard (per D-10)
    if (authUser.id === targetUserId) {
      throw new ForbiddenError(ErrorCodes.FORBIDDEN, "self_revocation");
    }

    const targetUser = await this.userRetriever.findByIdOrFail(targetUserId);

    // Super-user protection (per D-12): super user role is system-managed, cannot be revoked
    const hasSuperUserRole = targetUser.supportRoles?.items?.some(
      (role) => role.roleName === SupportRoleName.SUPER_USER,
    );
    if (hasSuperUserRole) {
      throw new ForbiddenError(ErrorCodes.FORBIDDEN, "system_managed_role");
    }

    // Last-admin protection (per D-11): prevent revoking if it would leave zero super users
    const superUserRole = await this.supportRoleRepository.findByRoleName(SupportRoleName.SUPER_USER);
    if (superUserRole) {
      const superUserCount = await this.userRepository.countBySupportRoleId(superUserRole.id);
      if (superUserCount <= 1) {
        throw new ForbiddenError(ErrorCodes.FORBIDDEN, "last_admin_protection");
      }
    }

    // Find the Admin role to get its ID for removal
    const adminRole = targetUser.supportRoles?.items?.find(
      (role) => role.roleName === SupportRoleName.SUPPORT_ADMINISTRATOR,
    );
    if (!adminRole) {
      throw new InvalidRequestError(ErrorCodes.INVALID_REQUEST, "User does not have the admin support role");
    }

    await this.userRepository.removeSupportRoleId(targetUserId, adminRole.id);
    this.logger.log("Revoked admin support role", { revokedBy: authUser.id, targetUserId, roleId: adminRole.id });

    return this.userRetriever.findByIdOrFail(targetUserId);
  }
}
```

Note on ForbiddenError: The `details` parameter is a string. Use `"self_revocation"`, `"system_managed_role"`, or `"last_admin_protection"` as the string value. The frontend will match on these string values to display appropriate error toasts.

Note on `supportRoles` access: Check `IUserDto` to see how hydrated roles are accessed. It may be `supportRoles` as a `DtoCollection` (with `.items` array), or a direct array. Adapt the `.some()` calls to match the actual shape. The PATTERNS.md shows `user.supportRoles.map(...)` in the controller response mapping, suggesting it may be iterable directly.

Note on ErrorCodes: Check `error-codes.enum.ts` for available values. Use `ErrorCodes.FORBIDDEN` for permission errors and `ErrorCodes.INVALID_REQUEST` for validation errors. If these enum values don't exist by those exact names, use the closest equivalent.
