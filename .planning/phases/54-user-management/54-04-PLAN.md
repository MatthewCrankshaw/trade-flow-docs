```typescript
import { apiSlice } from "@/services/api";
import type { PaginatedResponse, SupportUser } from "@/types";

interface SupportUsersParams {
  page?: number;
  limit?: number;
  search?: string;
  roleFilter?: string;
  subscriptionStatus?: string;
}

interface SupportUsersResult {
  users: SupportUser[];
  pagination: { total: number; page: number; limit: number };
}

export const supportApi = apiSlice.injectEndpoints({
  endpoints: (builder) => ({
    getSupportUsers: builder.query<SupportUsersResult, SupportUsersParams>({
      query: ({ page = 1, limit = 20, search, roleFilter, subscriptionStatus }) => {
        const params = new URLSearchParams();
        params.set("page", String(page));
        params.set("limit", String(limit));
        if (search) params.set("search", search);
        if (roleFilter && roleFilter !== "all") {
          params.set("filter:role.type:eq", roleFilter);
        }
        if (subscriptionStatus && subscriptionStatus !== "all") {
          params.set("filter:subscription.status:eq", subscriptionStatus);
        }
        return `/v1/support/users?${params.toString()}`;
      },
      transformResponse: (response: PaginatedResponse<SupportUser>) => ({
        users: response.data,
        pagination: response.pagination ?? { total: 0, page: 1, limit: 20 },
      }),
      providesTags: (result) =>
        result
          ? [
              ...result.users.map(({ id }) => ({ type: "SupportUser" as const, id })),
              { type: "SupportUser", id: "LIST" },
            ]
          : [{ type: "SupportUser", id: "LIST" }],
    }),
  }),
});

export const { useGetSupportUsersQuery } = supportApi;
```

**Create `src/features/support/hooks/useSupportUsers.ts`:**
This hook manages URL search param state for filters and pagination. Unlike `useCustomersList` which does client-side filtering, this hook drives server-side queries.

```typescript
import { useSearchParams } from "react-router-dom";
import { useMemo, useCallback, useRef, useEffect, useState } from "react";
import { useGetSupportUsersQuery } from "../api";

type RoleTab = "all" | "support" | "customer";
type SubscriptionFilter = "all" | "trialing" | "active" | "past_due" | "canceled" | "expired";

export function useSupportUsers() {
  const [searchParams, setSearchParams] = useSearchParams();
  const [searchInput, setSearchInput] = useState(searchParams.get("search") ?? "");
  const debounceTimerRef = useRef<ReturnType<typeof setTimeout>>();

  const page = Number(searchParams.get("page") ?? "1");
  const search = searchParams.get("search") ?? "";
  const roleFilter = (searchParams.get("filter:role.type:eq") ?? "all") as RoleTab;
  const subscriptionStatus = (searchParams.get("filter:subscription.status:eq") ?? "all") as SubscriptionFilter;

  const { data, isLoading, isFetching, error, refetch } = useGetSupportUsersQuery({
    page,
    limit: 20,
    search: search || undefined,
    roleFilter,
    subscriptionStatus,
  });

  // Debounced search: update URL param after 300ms
  const handleSearchChange = useCallback((value: string) => {
    setSearchInput(value);
    if (debounceTimerRef.current) clearTimeout(debounceTimerRef.current);
    debounceTimerRef.current = setTimeout(() => {
      setSearchParams((prev) => {
        const next = new URLSearchParams(prev);
        if (value) { next.set("search", value); } else { next.delete("search"); }
        next.set("page", "1");
        return next;
      });
    }, 300);
  }, [setSearchParams]);

  // Cleanup debounce timer
  useEffect(() => () => { if (debounceTimerRef.current) clearTimeout(debounceTimerRef.current); }, []);

  const setRoleFilter = useCallback((tab: RoleTab) => {
    setSearchParams((prev) => {
      const next = new URLSearchParams(prev);
      if (tab === "all") { next.delete("filter:role.type:eq"); } else { next.set("filter:role.type:eq", tab); }
      next.set("page", "1");
      return next;
    });
  }, [setSearchParams]);

  const setSubscriptionStatus = useCallback((status: SubscriptionFilter) => {
    setSearchParams((prev) => {
      const next = new URLSearchParams(prev);
      if (status === "all") { next.delete("filter:subscription.status:eq"); } else { next.set("filter:subscription.status:eq", status); }
      next.set("page", "1");
      return next;
    });
  }, [setSearchParams]);

  const setPage = useCallback((newPage: number) => {
    setSearchParams((prev) => {
      const next = new URLSearchParams(prev);
      next.set("page", String(newPage));
      return next;
    });
  }, [setSearchParams]);

  const totalPages = Math.ceil((data?.pagination.total ?? 0) / 20);

  return {
    users: data?.users ?? [],
    pagination: data?.pagination ?? { total: 0, page: 1, limit: 20 },
    totalPages,
    isLoading,
    isFetching,
    error,
    refetch,
    searchInput,
    handleSearchChange,
    roleFilter,
    setRoleFilter,
    subscriptionStatus,
    setSubscriptionStatus,
    page,
    setPage,
  };
}
```

```typescript
interface SubscriptionBadgeProps {

}

export function SubscriptionBadge({ status }: SubscriptionBadgeProps) {
  if (!status) {
    return <span className="inline-flex items-center rounded-md bg-muted px-2 py-1 text-xs font-medium text-muted-foreground">No subscription</span>;
  }

  const config: Record<string, { label: string; bg: string; text: string }> = {
    trialing: { label: "Trial", bg: "bg-amber-100", text: "text-amber-800" },
    active: { label: "Active", bg: "bg-green-100", text: "text-green-800" },
    past_due: { label: "Past Due", bg: "bg-orange-100", text: "text-orange-800" },
    canceled: { label: "Canceled", bg: "bg-muted", text: "text-muted-foreground" },
    expired: { label: "Expired", bg: "bg-muted", text: "text-muted-foreground" },
  };

  const { label, bg, text } = config[status] ?? { label: status, bg: "bg-muted", text: "text-muted-foreground" };

  return (
    <span className={cn("inline-flex items-center rounded-md px-2 py-1 text-xs font-medium", bg, text)}>
      {label}
    </span>
  );
}
```
