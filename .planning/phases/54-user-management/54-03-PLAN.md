**Create `src/support/services/firebase-auth-metadata.service.ts`:**
```typescript
import { Injectable } from "@nestjs/common";
import { ConfigService } from "@nestjs/config";
import * as admin from "firebase-admin";
import { AppLogger } from "@core/services/app-logger.service";

export interface IFirebaseAuthMetadata {
  creationTime: string | null;
  lastSignInTime: string | null;
}

@Injectable()
export class FirebaseAuthMetadataService {
  private readonly logger = new AppLogger(FirebaseAuthMetadataService.name);

  constructor(private readonly configService: ConfigService) {
    if (admin.apps.length === 0) {
      const projectId = this.configService.get<string>("FIREBASE_PROJECT_ID");
      const clientEmail = this.configService.get<string>("FIREBASE_CLIENT_EMAIL");
      const privateKey = this.configService.get<string>("FIREBASE_PRIVATE_KEY");

      if (clientEmail && privateKey) {
        admin.initializeApp({
          credential: admin.credential.cert({
            projectId,
            clientEmail,
            privateKey: privateKey.replace(/\\n/g, "\n"),
          }),
        });
      } else {
        admin.initializeApp({ projectId });
      }
    }
  }

  public async getAuthMetadata(externalAuthUserId: string): Promise<IFirebaseAuthMetadata> {
    try {
      const userRecord = await admin.auth().getUser(externalAuthUserId);
      return {
        creationTime: userRecord.metadata.creationTime ?? null,
        lastSignInTime: userRecord.metadata.lastSignInTime ?? null,
      };
    } catch (error) {
      this.logger.warn("Failed to fetch Firebase auth metadata", { externalAuthUserId, error });
      return { creationTime: null, lastSignInTime: null };
    }
  }
}
```

Key design decisions:
- Check `admin.apps.length === 0` before initializing to prevent "app already exists" error (Pitfall 4 from RESEARCH.md)
- Support both service account credentials (`FIREBASE_CLIENT_EMAIL` + `FIREBASE_PRIVATE_KEY`) and Application Default Credentials (just `FIREBASE_PROJECT_ID`)
- Gracefully return nulls on failure instead of throwing -- metadata is supplementary, not critical
- Replace `\\n` with actual newlines in private key (Railway stores multiline env vars with escaped newlines)

**Create `src/support/services/support-dashboard-metrics.service.ts`:**
`@Injectable()` class with `AppLogger`. Constructor injects `MongoDbFetcher`.

Method `getMetrics(): Promise<IDashboardMetricsDto>`:
- Execute aggregation pipeline on `"users"` collection:
  1. `$lookup` subscriptions (same pipeline-style as user list, with `$toString("$_id")` for type conversion)
  2. `$unwind` subscription (preserveNullAndEmptyArrays: true)
  3. `$group` with `_id: null` and conditional sums:
     - `totalUsers: { $sum: 1 }`
     - `activeTrials: { $sum: { $cond: [{ $eq: ["$subscriptionDocs.status", "trialing"] }, 1, 0] } }`
     - `activeSubscriptions: { $sum: { $cond: [{ $eq: ["$subscriptionDocs.status", "active"] }, 1, 0] } }`
     - `expiredSubscriptions: { $sum: { $cond: [{ $in: ["$subscriptionDocs.status", ["incomplete", "incomplete_expired"]] }, 1, 0] } }` (map "expired" display status to incomplete/incomplete_expired -- see RESEARCH.md open question 1; also count users with no subscription whose trial has ended if trialEnd is available)
     - `canceledSubscriptions: { $sum: { $cond: [{ $eq: ["$subscriptionDocs.status", "canceled"] }, 1, 0] } }`
- If result is empty, return all zeros
- Return `IDashboardMetricsDto`
