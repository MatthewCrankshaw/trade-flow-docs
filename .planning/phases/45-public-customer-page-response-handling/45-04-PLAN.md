2. **RTK Query API** (NEW `publicEstimateApi.ts`): Separate createApi instance with no auth headers per CUST-07. Mirror the `publicQuoteApi.ts` pattern exactly:
   ```typescript
   import { createApi, fetchBaseQuery } from "@reduxjs/toolkit/query/react";
   import type { PublicEstimate, RespondToEstimatePayload } from "../types/public-estimate.types";

   const publicBaseQuery = fetchBaseQuery({
     baseUrl: import.meta.env.VITE_API_BASE_URL || "http://localhost:3000",
   });

   export const publicEstimateApi = createApi({
     reducerPath: "publicEstimateApi",
     baseQuery: publicBaseQuery,
     tagTypes: ["PublicEstimate"],
     endpoints: (builder) => ({
       getPublicEstimate: builder.query<PublicEstimate, string>({
         query: (token) => `/v1/public/estimate/${token}`,
         transformResponse: (response: { data: PublicEstimate[] }) => response.data[0],
         providesTags: ["PublicEstimate"],
       }),
       respondToEstimate: builder.mutation<PublicEstimate, { token: string; body: RespondToEstimatePayload }>({
         query: ({ token, body }) => ({
           url: `/v1/public/estimate/${token}/respond`,
           method: "POST",
           body,
         }),
         transformResponse: (response: { data: PublicEstimate[] }) => response.data[0],
         invalidatesTags: ["PublicEstimate"],
       }),
     }),
   });

   export const { useGetPublicEstimateQuery, useRespondToEstimateMutation } = publicEstimateApi;
   ```
