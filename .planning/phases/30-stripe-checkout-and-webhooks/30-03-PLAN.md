    Implementation:
    ```typescript
    async verifySession(sessionId: string, authUser: IUserDto): Promise<IVerifySessionResponse> {
      // Fast path: check local record by checkout session ID
      const subscription = await this.subscriptionRepository.findByCheckoutSessionId(sessionId);

      if (
        subscription &&
        (subscription.status === SubscriptionStatus.TRIALING || subscription.status === SubscriptionStatus.ACTIVE)
      ) {
        this.logger.log("verify-session: found local record", {
          sessionId,
          status: subscription.status,
        });
        return { status: subscription.status, verified: true };
      }

      // Fallback: webhook hasn't fired yet, query Stripe API directly
      this.logger.log("verify-session: falling back to Stripe API", { sessionId });
      try {
        const session = await this.stripe.checkout.sessions.retrieve(sessionId);
        return { status: session.payment_status, verified: true };
      } catch (error) {
        this.logger.error("verify-session: Stripe API call failed", error.stack, { sessionId });
        throw error;
      }
    }
    ```

    Import `Stripe` from `"stripe"`, `Inject` from `@nestjs/common`, `STRIPE_CLIENT` from the provider, `SubscriptionRepository` from the repository, `SubscriptionStatus` from the enum, `IVerifySessionResponse` from responses, and the auth user interface from `@auth/interfaces/`.
