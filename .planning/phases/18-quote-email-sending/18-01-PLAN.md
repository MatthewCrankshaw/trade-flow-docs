```typescript
import { Injectable } from "@nestjs/common";
import { ConfigService } from "@nestjs/config";
import * as fs from "fs";
import * as path from "path";

export interface QuoteEmailData {
  businessName: string;
  message: string;
  viewUrl: string;
  quoteNumber: string;
  quoteDate: string;
  total: string;
}

@Injectable()
export class QuoteEmailRenderer {
  constructor(private readonly configService: ConfigService) {}

  public async render(data: QuoteEmailData): Promise<string> {
    const templatePath = path.join(__dirname, "../templates/quote-email.html");
    const template = fs.readFileSync(templatePath, "utf-8");

    const populated = template
      .replace(/\{\{\s*businessName\s*\}\}/g, this.escapeHtml(data.businessName))
      .replace(/\{\{\{\s*message\s*\}\}\}/g, data.message) // Triple braces = raw HTML (sanitized before this point)
      .replace(/\{\{\s*viewUrl\s*\}\}/g, data.viewUrl)
      .replace(/\{\{\s*quoteNumber\s*\}\}/g, this.escapeHtml(data.quoteNumber))
      .replace(/\{\{\s*quoteDate\s*\}\}/g, this.escapeHtml(data.quoteDate))
      .replace(/\{\{\s*total\s*\}\}/g, this.escapeHtml(data.total));

    // Use dynamic import for ESM-only Maizzle package
    const { render } = await (Function('return import("@maizzle/framework")')() as Promise<typeof import("@maizzle/framework")>);
    const result = await render(populated, {
      tailwind: {
        config: {
          content: [{ raw: populated, extension: "html" }],
        },
      },
    });
    return result.html;
  }

  private escapeHtml(text: string): string {
    return text
      .replace(/&/g, "&amp;")
      .replace(/</g, "&lt;")
      .replace(/>/g, "&gt;")
      .replace(/"/g, "&quot;");
  }
}
```

NOTE: The ESM import approach for Maizzle v5 (ESM-only) in a CommonJS NestJS project needs dynamic import(). Use `Function('return import("@maizzle/framework")')()` to avoid TypeScript transpiling import() to require(). If this doesn't work at runtime, fall back to a simpler approach: read the template, do string replacements, and inline CSS manually without Maizzle (the template already uses inline styles as fallback).

```typescript
@UseGuards(JwtAuthGuard)
@Post("business/:businessId/quote/:quoteId/send")
public async send(
  @Req() request: { user: IUserDto; params: { businessId: string; quoteId: string } },
  @Body() body: SendQuoteRequest,
): Promise<IResponse<IQuoteResponse>> {
  try {
    const { businessId, quoteId } = request.params;
    const updatedQuote = await this.quoteEmailSender.send(request.user, quoteId, businessId, body);
    const response = await this.enrichAndMapToResponse(request.user, updatedQuote);
    return createResponse([response]);
  } catch (error) {
    const httpError = createHttpError(error);
    throw httpError;
  }
}
```
