```typescript
import { Resend } from "resend";
import { HttpException, HttpStatus, Injectable } from "@nestjs/common";
import { ConfigService } from "@nestjs/config";
import { SendEmailDto } from "@email/interfaces/send-email.dto";
import { AppLogger } from "@core/services/app-logger.service";

@Injectable()
export class EmailSenderService {
  private readonly logger = new AppLogger(EmailSenderService.name);
  private readonly apiKeyConfigured: boolean;
  private readonly resend: Resend | null;

  constructor(private readonly configService: ConfigService) {
    const apiKey = this.configService.get<string>("RESEND_API_KEY");
    const placeholderPatterns = ["your-", "placeholder"];
    const isPlaceholder =
      !apiKey || placeholderPatterns.some((p) => apiKey.toLowerCase().includes(p)) || apiKey === "your-resend-api-key-here";

    if (isPlaceholder) {
      this.logger.warn("Resend API key is missing or appears to be a placeholder. Email sending will be unavailable.");
      this.apiKeyConfigured = false;
      this.resend = null;
    } else {
      this.resend = new Resend(apiKey);
      this.apiKeyConfigured = true;
    }
  }

  public async sendEmail(email: SendEmailDto): Promise<void> {
    if (!this.apiKeyConfigured || !this.resend) {
      throw new HttpException("Email sending is not configured. Set a valid RESEND_API_KEY environment variable.", HttpStatus.SERVICE_UNAVAILABLE);
    }

    const { data, error } = await this.resend.emails.send({
      from: `${email.senderName} <${email.from}>`,
      to: [email.to],
      subject: email.subject,
      html: email.html,
    });

    if (error) {
      this.logger.error("Resend send failed", { to: email.to, error: error.message });
      throw new HttpException(`Email delivery failed: ${error.message}`, HttpStatus.BAD_GATEWAY);
    }

    this.logger.log("Email sent successfully", { to: email.to, id: data?.id });
  }
}
```
