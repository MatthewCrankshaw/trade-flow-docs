    **Step 2: Create `e2e/helpers/api-client.ts` — Typed HTTP seeding client (per D-01, D-02).**

    The client wraps fetch() calls to existing NestJS REST API endpoints. It receives the base URL and Bearer token at construction time. Each method maps to one POST endpoint and returns the created entity. The token comes from the live page context (not from storageState JSON parsing) to avoid Firebase token expiration issues (Pitfall 1 from RESEARCH.md).

    ```typescript
    // e2e/helpers/api-client.ts
    export interface SeededCustomer {
      id: string;
      name: string;
      email: string;
      businessId: string;
    }

    export interface SeededJob {
      id: string;
      title: string;
      customerId: string;
      businessId: string;
    }

    export interface SeededQuote {
      id: string;
      jobId: string;
      businessId: string;
    }

    export class ApiClient {
      constructor(
        private readonly baseUrl: string,
        private readonly token: string,
        private readonly businessId: string,
      ) {}

      private async post<T>(path: string, body: Record<string, unknown>): Promise<T> {
        const res = await fetch(`${this.baseUrl}${path}`, {
          method: "POST",
          headers: {
            "Content-Type": "application/json",
            Authorization: `Bearer ${this.token}`,
          },
          body: JSON.stringify(body),
        });
        if (!res.ok) {
          throw new Error(`API seed failed: ${res.status} ${await res.text()} — POST ${path}`);
        }
        const json = await res.json();
        return json.data[0];
      }

      private async get<T>(path: string): Promise<T[]> {
        const res = await fetch(`${this.baseUrl}${path}`, {
          headers: { Authorization: `Bearer ${this.token}` },
        });
        if (!res.ok) {
          throw new Error(`API fetch failed: ${res.status} ${await res.text()} — GET ${path}`);
        }
        const json = await res.json();
        return json.data;
      }

      async createCustomer(payload: { name: string; email: string; phone: string; businessId: string }): Promise<SeededCustomer> {
        return this.post<SeededCustomer>("/v1/customer", payload);
      }

      async createJob(payload: { title: string; customerId: string; businessId: string }): Promise<SeededJob> {
        return this.post<SeededJob>("/v1/job", payload);
      }

      async createQuote(payload: { jobId: string; businessId: string }): Promise<SeededQuote> {
        return this.post<SeededQuote>("/v1/quote", payload);
      }

      async getBusinesses(): Promise<Array<{ id: string; name: string }>> {
        return this.get<{ id: string; name: string }>("/v1/business");
      }

      async getJobTypes(businessId: string): Promise<Array<{ id: string; name: string }>> {
        return this.get<{ id: string; name: string }>(`/v1/job-type?businessId=${businessId}`);
      }

      async getTaxRates(businessId: string): Promise<Array<{ id: string; name: string }>> {
        return this.get<{ id: string; name: string }>(`/v1/tax-rate?businessId=${businessId}`);
      }

      async getItems(businessId: string): Promise<Array<{ id: string; name: string }>> {
        return this.get<{ id: string; name: string }>(`/v1/item?businessId=${businessId}`);
      }
    }
    ```

    **Step 3: Create `e2e/helpers/db-cleanup.ts` — MongoDB cleanup helper (per D-05).**

    Connects directly to MongoDB via the native driver. Deletes documents by `businessId` across known collections. Does NOT delete from `users`, `business`, or `business_users` (those are fixture data). The onboarding test has its own specialized cleanup (Plan 02).

    ```typescript
    // e2e/helpers/db-cleanup.ts
    import { MongoClient, type Db } from "mongodb";

    // Collections safe to clean by businessId filter.
    // Source: INTEGRATIONS.md collections list.
    // Do NOT add: users, business, business_users (test fixture data).
    const CLEANUP_COLLECTIONS = [
      "customers",
      "jobs",
      "quotes",
      "quote_line_items",
      "items",
      "tax_rates",
      "job_types",
    ];

    export class DbCleanup {
      private client: MongoClient;
      private db: Db;

      constructor(mongoUrl: string, dbName = "trade-flow-db") {
        this.client = new MongoClient(mongoUrl);
        this.db = this.client.db(dbName);
      }

      async connect(): Promise<void> {
        await this.client.connect();
      }

      async cleanByBusinessId(businessId: string): Promise<void> {
        for (const collection of CLEANUP_COLLECTIONS) {
          await this.db.collection(collection).deleteMany({ businessId });
        }
      }

      /** Onboarding-specific cleanup: removes the business + defaults created by the wizard. */
      async cleanOnboardingData(businessId: string, userId: string): Promise<void> {
        // Delete all standard collections by businessId
        await this.cleanByBusinessId(businessId);
        // Delete the business itself
        await this.db.collection("business").deleteMany({ _id: businessId as any });
        // Delete business_users link
        await this.db.collection("business_users").deleteMany({ businessId });
        // Clear businessIds from user document (don't delete the user)
        await this.db.collection("users").updateMany(
          { firebaseUid: userId },
          { $set: { businessIds: [] } },
        );
      }

      async close(): Promise<void> {
        await this.client.close();
      }
    }
    ```

    **Step 4: Create `e2e/helpers/seed-factories.ts` — Faker factory functions (per D-03, D-07).**

    Each factory returns a payload object matching the API's expected request body. Uses `@faker-js/faker` for realistic values and `Date.now()` suffix for uniqueness.

    ```typescript
    // e2e/helpers/seed-factories.ts
    import { faker } from "@faker-js/faker";

    export function customerPayload(businessId: string) {
      const ts = Date.now();
      return {
        name: `${faker.person.fullName()} ${ts}`,
        email: `test-${ts}@example.com`,
        phone: faker.phone.number(),
        businessId,
      };
    }

    export function jobPayload(businessId: string, customerId: string) {
      const ts = Date.now();
      return {
        title: `${faker.commerce.productName()} ${ts}`,
        customerId,
        businessId,
      };
    }

    export function quotePayload(businessId: string, jobId: string) {
      return {
        jobId,
        businessId,
      };
    }
    ```

    import { ApiClient } from "../helpers/api-client";
    import { DbCleanup } from "../helpers/db-cleanup";

    async function extractTokenFromPage(page: Page): Promise<string> {
      return page.evaluate(async () => {
        const { getAuth } = await import("firebase/auth");
        const auth = getAuth();
        const user = auth.currentUser;
        if (!user) throw new Error("No authenticated user in browser context");
        return user.getIdToken();
      });

    type TestFixtures = {
      apiClient: ApiClient;
      dbCleanup: DbCleanup;

        await page.waitForLoadState("networkidle");

        const mongoUrl = process.env.E2E_MONGO_URL ?? "mongodb://localhost:27017";

    setup("authenticate onboarding user", async ({ page }) => {
      // Same API health check is not needed again — already validated in first setup
      await page.goto("/");

      await page.evaluate(
        async ({ email, password, firebaseConfig }) => {
          const { initializeApp, getApps } = await import("firebase/app");
          const { getAuth, signInWithEmailAndPassword } = await import("firebase/auth");
          const app = getApps().length > 0 ? getApps()[0] : initializeApp(firebaseConfig);
          const auth = getAuth(app);
          // Sign out any existing user first (previous setup may have signed in the pre-onboarded user)
          await auth.signOut();
          await signInWithEmailAndPassword(auth, email, password);
        },
        {
          email: process.env.E2E_ONBOARDING_USER_EMAIL!,
          password: process.env.E2E_ONBOARDING_USER_PASSWORD!,
          firebaseConfig: {
            apiKey: process.env.E2E_FIREBASE_API_KEY,
            authDomain: process.env.E2E_FIREBASE_AUTH_DOMAIN,
            projectId: process.env.E2E_FIREBASE_PROJECT_ID,

      // Onboarding user has no business — the app should show the onboarding wizard.

      await page.context().storageState({ path: ONBOARDING_AUTH_FILE, indexedDB: true });
