    import { test, expect } from "../../fixtures/test-base";

    // Override storageState to use the onboarding user (no business)
    test.use({ storageState: "e2e/.auth/onboarding-user.json" });

    // Helper: extract Firebase token from page context
    async function extractTokenFromPage(page: Page): Promise<string> {
      return page.evaluate(async () => {
        const { getAuth } = await import("firebase/auth");
        const auth = getAuth();
        const user = auth.currentUser;
        if (!user) throw new Error("No authenticated user in browser context");
        return user.getIdToken();
      });
    }

    // Helper: extract Firebase UID from page context
    async function extractUidFromPage(page: Page): Promise<string> {
      return page.evaluate(async () => {
        const { getAuth } = await import("firebase/auth");
        return getAuth().currentUser!.uid;
      });
    }

    // Helper: retry API call until non-empty result (Pitfall 5 avoidance)
    async function waitForDefaults<T>(
      fn: () => Promise<T[]>,
      description: string,
      retries = 5,
      delayMs = 500,
    ): Promise<T[]> {
      for (let i = 0; i < retries; i++) {
        const result = await fn();
        if (result.length > 0) return result;
        await new Promise((r) => setTimeout(r, delayMs));
      }
      throw new Error(`${description} — no results after ${retries} retries`);
    }

    test.describe("AUTH-03: Onboarding wizard", () => {
      let createdBusinessId: string | null = null;
      let userId: string | null = null;

      test.afterAll(async () => {
        // Comprehensive cleanup per Pitfall 3 in RESEARCH.md
        if (createdBusinessId && userId) {
          const mongoUrl = process.env.E2E_MONGO_URL ?? "mongodb://localhost:27017";
          const cleanup = new DbCleanup(mongoUrl);
          await cleanup.connect();
          await cleanup.cleanOnboardingData(createdBusinessId, userId);
          await cleanup.close();
        }
      });

      test("completes wizard and creates business with defaults", async ({ page }) => {
        const businessName = `Test Plumbing ${Date.now()}`;

        // Navigate to app — onboarding wizard should appear for user with no business
        await page.goto("/");

        // --- WIZARD STEPS ---
        // IMPORTANT: The exact selectors below are TEMPLATES based on RESEARCH.md guidance.
        // The executor MUST adjust selectors after reading the actual onboarding component source.
        // Use ARIA role selectors first (getByRole, getByLabel). Add data-testid only if ambiguous.

        // Step: Enter business name
        // Look for a text input labeled "Business name" or similar
        await page.getByLabel(/business name/i).fill(businessName);

        // Step: Select trade
        // This may be a Radix Select/Combobox — adjust selector based on component source
        await page.getByRole("combobox", { name: /trade/i }).click();
        await page.getByRole("option", { name: /plumb/i }).click();

        // Step: Complete wizard
        // Click the final submit/complete/continue button
        await page.getByRole("button", { name: /complete|finish|continue|save/i }).click();

        // D-09 Assertion 3: User redirected to the dashboard after completing onboarding
        await expect(page).toHaveURL(/\/dashboard/, { timeout: 15000 });

        // --- API VERIFICATION (D-09 Assertions 1 and 2) ---
        const token = await extractTokenFromPage(page);
        userId = await extractUidFromPage(page);
        const apiUrl = process.env.E2E_API_URL ?? "http://localhost:3000";
        const apiClient = new ApiClient(apiUrl, token, "");

        // D-09 Assertion 1: Business created with correct fields
        const businesses = await apiClient.getBusinesses();
        expect(businesses.length).toBeGreaterThanOrEqual(1);
        const business = businesses.find((b) => b.name === businessName);
        expect(business).toBeDefined();
        createdBusinessId = business!.id;

        // D-09 Assertion 2: Default entities exist — verified via API calls (not UI navigation)
        // Use retry loop to handle Pitfall 5 (race condition)
        const jobTypes = await waitForDefaults(
          () => apiClient.getJobTypes(createdBusinessId!),
          "Expected default job types after onboarding",
        );
        expect(jobTypes.length).toBeGreaterThanOrEqual(1);

        const taxRates = await waitForDefaults(
          () => apiClient.getTaxRates(createdBusinessId!),
          "Expected default tax rates after onboarding",
        );
        expect(taxRates.length).toBeGreaterThanOrEqual(1);

        const items = await waitForDefaults(
          () => apiClient.getItems(createdBusinessId!),
          "Expected default items after onboarding",
        );
        expect(items.length).toBeGreaterThanOrEqual(1);
      });

      test("subsequent login skips onboarding wizard", async ({ page }) => {
        // D-09 Assertion 4: Log out, log back in, lands on dashboard (not onboarding)
        // This test runs after "completes wizard" so the onboarding user now has a business

        await page.goto("/");

        // Should land on dashboard directly — no onboarding wizard
        await expect(page).toHaveURL(/\/dashboard/, { timeout: 15000 });
        await expect(page.getByRole("main")).toBeVisible();
      });
    });
