   ```typescript
   private async handleImpersonationToken(token: string, request: Request): Promise<boolean> {
     try {
       const payload = jwt.verify(token, this.configService.get<string>("IMPERSONATION_JWT_SECRET"), {
         algorithms: ["HS256"],
         issuer: "trade-flow-impersonation",
       }) as IImpersonationPayload;

       const targetUser = await this.userRetriever.getById(payload.targetUserId);
       if (!targetUser) {
         throw new UnauthorizedException("Target user not found");
       }

       const supportUser = await this.userRetriever.getById(payload.supportUserId);
       if (!supportUser) {
         throw new UnauthorizedException("Support user not found");
       }

       request.user = targetUser;
       request.impersonator = supportUser;
       return true;
     } catch (error) {
       if (error instanceof jwt.TokenExpiredError) {
         throw new UnauthorizedException("Impersonation session expired. Please start a new session.");
       }
       if (error instanceof jwt.JsonWebTokenError) {
         throw new UnauthorizedException("Invalid impersonation token");
       }
       throw error;
     }
   }
   ```

```typescript
import { Controller, Post, Body, Req, UseGuards } from "@nestjs/common";
import { JwtAuthGuard } from "@auth/auth.guard";
import { IUserDto } from "@user/data-transfer-objects/user.dto";
import { IResponse } from "@core/response/response.interface";
import { createResponse } from "@core/response/create-response.utility";
import { createHttpError } from "@core/errors/handle-error.utility";
import { ImpersonationCreator } from "@impersonation/services/impersonation-creator.service";
import { ImpersonationTerminator } from "@impersonation/services/impersonation-terminator.service";
import { StartImpersonationRequest } from "@impersonation/requests/start-impersonation.request";
import { EndImpersonationRequest } from "@impersonation/requests/end-impersonation.request";
import { IImpersonationResponse } from "@impersonation/responses/impersonation.response";

@Controller("v1")
export class ImpersonationController {
  constructor(
    private readonly impersonationCreator: ImpersonationCreator,
    private readonly impersonationTerminator: ImpersonationTerminator,
  ) {}

  @UseGuards(JwtAuthGuard)
  @Post("impersonation/start")
  public async start(
    @Req() request: { user: IUserDto },
    @Body() body: StartImpersonationRequest,
  ): Promise<IResponse<IImpersonationResponse>> {
    try {
      const result = await this.impersonationCreator.create(request.user, body);
      return createResponse([result]);
    } catch (error) {
      throw createHttpError(error);
    }
  }

  @UseGuards(JwtAuthGuard)
  @Post("impersonation/end")
  public async end(
    @Body() body: EndImpersonationRequest,
  ): Promise<IResponse<void>> {
    try {
      await this.impersonationTerminator.end(body.sessionId);
      return createResponse([]);
    } catch (error) {
      throw createHttpError(error);
    }
  }
}
```
