**derivePaywallVariant function (per D-05):**

Define inside PaywallGuard.tsx or as a separate utility:

```typescript
type PaywallVariant = "trial-expired" | "payment-failed" | "canceled";

function derivePaywallVariant(subscription: SubscriptionType | null): PaywallVariant {
  if (!subscription) return "trial-expired"; // No record = fallback

  if (subscription.status === "past_due") return "payment-failed";

  if (subscription.status === "canceled") {
    const trialEnd = new Date(subscription.trialEnd).getTime();
    const canceledAt = new Date(subscription.canceledAt).getTime();
    const TWENTY_FOUR_HOURS = 24 * 60 * 60 * 1000;
    if (Math.abs(canceledAt - trialEnd) <= TWENTY_FOUR_HOURS) {
      return "trial-expired";
    }
    return "canceled";
  }

  return "trial-expired"; // Fallback for unexpected status
}
```
