```typescript
// findAll pattern:
@Get("business/:businessId/jobs")
public async findAll(@Req() request: { user: IUserDto; params: { businessId: string }; query: Record<string, string | string[] | undefined> }) {
  const queryOptions = paginationQueryToBaseQueryOptions(request.query);
  const collection = await this.jobRetriever.findAllByBusinessId(request.user, businessId, queryOptions);
  const responses = collection.map((dto) => mapJobToResponse(dto));
  return createResponse(responses, collection.queryResults);
}

// findOne pattern:
@Get("job/:jobId")
public async findOne(@Req() request: { user: IUserDto; params: { jobId: string } }) {
  const job = await this.jobRetriever.findByIdOrFail(request.user, jobId);
  const response = mapJobToResponse(job);
  return createResponse([response]);
}

// update pattern:
@Patch("business/:businessId/job/:jobId")
public async update(@Req() request, @Body() requestBody: UpdateJobRequest) {
  const existing = await this.jobRetriever.findByIdOrFail(request.user, jobId);
  const merged = mergeExistingJobWithChanges(existing, requestBody);
  const updated = await this.jobUpdater.update(request.user, existing, merged);
  return createResponse([mapJobToResponse(updated)]);
}
```
