```typescript
import { Injectable } from "@nestjs/common";
import { InjectQueue } from "@nestjs/bullmq";
import { Queue } from "bullmq";
import { QUEUE_NAMES } from "@queue/queue.constant";
import { AppLogger } from "@core/services/app-logger.service";
import { FOLLOWUP_DELAYS_MS, followupJobId, expiryJobId } from "@estimate-followups/services/followup-delays.constant";
import type { IFollowupJobData, IExpiryJobData } from "@estimate-followups/data-transfer-objects/followup-job.dto";

@Injectable()
export class EstimateFollowupScheduler {
  private readonly logger = new AppLogger(EstimateFollowupScheduler.name);

  constructor(
    @InjectQueue(QUEUE_NAMES.ESTIMATE_FOLLOWUPS)
    private readonly queue: Queue,
  ) {}

  public async scheduleFollowups(estimateId: string, revisionNumber: number): Promise<void> {
    const steps: Array<{ step: 1 | 2 | 3; delay: number }> = [
      { step: 1, delay: FOLLOWUP_DELAYS_MS.STEP_1 },
      { step: 2, delay: FOLLOWUP_DELAYS_MS.STEP_2 },
      { step: 3, delay: FOLLOWUP_DELAYS_MS.STEP_3 },
    ];

    await Promise.all(
      steps.map(({ step, delay }) => {
        const data: IFollowupJobData = { estimateId, revisionNumber, step };
        return this.queue.add("follow-up", data, {
          jobId: followupJobId(estimateId, revisionNumber, step),
          delay,
        });
      }),
    );

    const expiryData: IExpiryJobData = { estimateId, revisionNumber };
    await this.queue.add("expiry", expiryData, {
      jobId: expiryJobId(estimateId, revisionNumber),
      delay: FOLLOWUP_DELAYS_MS.EXPIRY,
    });

    this.logger.log("Scheduled 3 follow-ups + 1 expiry job", { estimateId, revisionNumber });
  }
}
```

```typescript
import { Injectable } from "@nestjs/common";
import { InjectQueue } from "@nestjs/bullmq";
import { Queue } from "bullmq";
import { QUEUE_NAMES } from "@queue/queue.constant";
import { AppLogger } from "@core/services/app-logger.service";
import type { IEstimateFollowupCanceller } from "@estimate/services/estimate-followup-canceller.interface";
import { followupJobId, expiryJobId } from "@estimate-followups/services/followup-delays.constant";

@Injectable()
export class BullMQEstimateFollowupCanceller implements IEstimateFollowupCanceller {
  private readonly logger = new AppLogger(BullMQEstimateFollowupCanceller.name);

  constructor(
    @InjectQueue(QUEUE_NAMES.ESTIMATE_FOLLOWUPS)
    private readonly queue: Queue,
  ) {}

  public async cancelAllFollowups(estimateId: string, revisionNumber: number): Promise<void> {
    const jobIds = [
      followupJobId(estimateId, revisionNumber, 1),
      followupJobId(estimateId, revisionNumber, 2),
      followupJobId(estimateId, revisionNumber, 3),
      expiryJobId(estimateId, revisionNumber),
    ];

    await Promise.all(
      jobIds.map(async (id) => {
        try {
          await this.queue.remove(id);
        } catch (error) {
          this.logger.error(`Failed to remove job ${id}`, (error as Error).stack);
        }
      }),
    );

    this.logger.log("Cancelled follow-up/expiry jobs", { estimateId, revisionNumber, jobCount: jobIds.length });
  }
}
```
